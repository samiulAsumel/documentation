# Where am I?

pwd

# Go somewhere

cd /path/to/dir # absolute
cd ..           # up one level
cd ~            # home
cd -            # previous dir

# List files

ls      # basic
ls -l   # long format
ls -la  # + hidden files
ls -lh  # + human readable
ls -lt  # + sorted by time
ls -lR  # recursive

# File type in ls -l output

# - = file | d = directory | l = symlink

# System info

whoami              # current user
id                  # user + group details
cat /etc/os-release # OS version
hostname            # system hostname
uname -r            # kernel version

# ==========================================

# === TAB COMPLETION ===
<partial><TAB>        # complete করো
<partial><TAB><TAB>   # সব options দেখো

# === HISTORY ===
history               # সব commands দেখো
!!                    # last command repeat
!n                    # history number n run করো
!string               # "string" দিয়ে শুরু last command
Ctrl+R                # reverse search (most useful!)

# === KEYBOARD SHORTCUTS ===
Ctrl+A / Ctrl+E       # line শুরু / শেষ
Ctrl+U / Ctrl+K       # বাম / ডান দিক delete
Ctrl+W                # one word delete (left)
Ctrl+L                # clear screen
Ctrl+C                # kill running command

# === VARIABLES ===
VAR="value"           # shell variable
export VAR            # environment variable বানাও
echo $VAR             # use করো
env | grep VAR        # verify
unset VAR             # delete করো
echo $?               # last command exit code

# === ALIASES ===
alias name='command'  # temporary alias
echo "alias..." >> ~/.bashrc && source ~/.bashrc  # permanent

# === IMPORTANT BUILT-INS ===
echo $PATH $HOME $USER $SHELL $PWD $HOSTNAME $$
